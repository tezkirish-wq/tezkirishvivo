const SHEET_ID      = 'ВАШ_SHEET_ID';
const SHEET_MAIN    = 'Активности';
const SHEET_COMPANY = 'Компании';

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let cs = ss.getSheetByName(SHEET_COMPANY);
  if (!cs) cs = ss.insertSheet(SHEET_COMPANY);
  const last = cs.getLastRow();
  let companies = [];
  if (last > 0) {
    const data = cs.getRange(1,1,last,2).getValues();
    companies = data
      .filter(r => r[0])
      .map(r => ({ name: String(r[0]).trim(), country: String(r[1]).trim() }))
      .sort((a,b) => a.name.localeCompare(b.name, ['ru','en']));
  }
  return ContentService
    .createTextOutput(JSON.stringify({ companies }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const d  = JSON.parse(e.postData.contents);

  if (d.action === 'addCompany') {
    let cs = ss.getSheetByName(SHEET_COMPANY);
    if (!cs) cs = ss.insertSheet(SHEET_COMPANY);
    cs.appendRow([d.name, d.country]);
    return ContentService
      .createTextOutput(JSON.stringify({ result: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  let ms = ss.getSheetByName(SHEET_MAIN);
  if (!ms) ms = ss.insertSheet(SHEET_MAIN);
  if (ms.getLastRow() === 0) {
    ms.appendRow([
      'Отметка времени (ввод)',
      'Дата события',
      'ФИО сотрудника',
      'Этап воронки',
      'Название компании',
      'Страна компании',
      'Комментарий'
    ]);
    ms.getRange(1,1,1,7)
      .setFontWeight('bold')
      .setBackground('#0F6E56')
      .setFontColor('#ffffff');
    ms.setFrozenRows(1);
  }

  d.stages.forEach(stage => {
    ms.appendRow([
      d.timestamp, d.actionDate, d.employee,
      stage, d.company, d.country, d.comment
    ]);
  });

  return ContentService
    .createTextOutput(JSON.stringify({ result: 'ok' }))
    .setMimeType(ContentService.MimeType.JSON);
}
